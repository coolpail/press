# 先看Promise 构造函数

class MyPromise {
    //传入一个函数
    constructor (handler) {
        if (typeof handler !== 'Function') {
            throw new Error('handler is not function')
        }
        //定义状态
        this.status = 'PENDING';
        //定义值
        this._value = undefined;
        // 添加成功回调函数队列
        this._fulfilledQueues = []
        // 添加失败回调函数队列
        this._rejectedQueues = []
        try {
            handler(this._resolve.bind(this), this._reject.bind(this));
        } catch(err) {
            this._reject(err)
        }
        _resolve(val) {
            const run = () => {
                if (this._status !== 'PENDING') return;

                        runFulfilled(res);

                    runFulfilled(res);

            };

            setTimeOut(run, 0);
        },
        _rejectt(err) {
            const run = () => {
                this.status = 'REJECTED';
                this._value = err;
                let cb;

                    cb(err);
                }
            };
            setTimeOut(run, 0);
        }
    }
}

总结

  • handler函数会被执行，resolve和reject函数会异步调用；
  • 为什么bind(this)，因为绑定函数单独执行this是undefined
  • handler执行报错也会被catch然后执行reject
  • resolve入参是Promsie时，会等待其状态的改变，所以resovle和reject都要写入

# then方法

注意点

  • then入参不是函数时，直接用前一个Promsie的状态改变时带入的值，如果是函数，会带入val参数执行函数，需要看结果是否是promsie，不是则直接resolve或者reject改值，是的话则需要等待其状态改变在then中入参才带入之前的需要执行resolve或者reject，因为这是改变当前Promise的状态
  • 状态为pending时需要将then中的入参函数加入对应的回调数组中_fulfilledQueues,_rejectedQueues，等待状态的改变然后遍历数组，如果状态已经改变就直接执行对应then中的函数
  • then的函数默认返回一个Promise
MyPromise.prototype.then = (onFulfilled, onRejected) => {
    const { _value, status } = this;
    return new MyPromise((onFulfilledNext, onRejectedNext) => {
        let fulfilled = (value) => {
            try {
                if (typeof onFulfilled !== 'function') {
                    onFulfilledNext(value);
                } else {
                    let result = onFulfilled(value);
                    if (result instanceof MyPromise) {
                        result.then(onFulfilledNext, onRejectedNext);
                    } else {
                        onFulfilledNext(result);
                    }
                } 
            } catch (err) {
                onRejectedNext(err);
            }
        };
        let rejected = (value) => {
            try {
                if (typeof onFulfilled !== 'function') {
                    onRejectedNext(value);
                } else {
                    let result = onRejected(value);
                    if (result instanceof MyPromise) {
                        result.then(onFulfilledNext, onRejectedNext);
                    } else {
                        //**这边需要注意，因为reject(val)后接then()是可以的，所以这边是onFulfilledNext**
                        onFulfilledNext(result);
                    }
                } 
            } catch (err) {
                onRejectedNext(err);
            }
        };
        switch (status) {
            case 'PENDING':
                this._fulfilledQueues.push(fulfilled);
                this._rejectedQueues.push(rejected);
                break;
            // 当状态已经改变时，立即执行对应的回调函数
            case FULFILLED:
                fulfilled(_value)
                break;
            case REJECTED:
                rejected(_value)
                break;
        }
    })
}

# catch方法

MyPromise.prototype.catch = (onRejected) => {
    return this.then(null, onRejected); //需要return，不然不能链路
}
  • 将参数带入到then的第二个参数中，只是语法糖，建议还是用then

# finally方法

MyPromise.prototype.finally = (cb) => {
    return this.then(
        value => MyPromise.resolve(cb()).then(() => value);
        err => MyPromise.resolve(cb()).then(() => { throw err });
    )
}
  • 可以看出finally，最后得到的结果还是之前的value，但是cb是一定会被执行的

# 静态方法resolve

MyPromise.resolve = (val) => {
    //是promise等待val的then执行
    if (val instanceof MyPromise) return val;
    return new MyPromise(res => res(val))
}

# 静态方法reject

MyPromise.reject = (err) => {
    //是promise等待val的then执行
    return new MyPromise((res, rej) => rej(err));
}
  • 这边reject的静态方法的入参不判断是否是promise, 是页直接带入待下个函数作为入参

# 静态方法all

MyPromise.all = (list) => {
    return new MyPromise((res, rej) => {
        let value = [];
        let count = 0;
        list.forEach((el, index) => {
            //el可以不是promise对象，所以要进行promise初始化
            this.resolve(el).then(
                (val) => {
                    value[index] = val;
                    count++;
                    if (count === list.length) res(value);
                },
                (err) => {
                    rej(err);
                }
            )
        })
    })
}

# 静态方法race

MyPromise.race = (list) => {
    return new MyPromise((res, rej) => {
        list.forEach((el, index) => {
            // 只要有一个实例率先改变状态，新的MyPromise的状态就跟着改变
            this.resolve(el).then(
                (val) => {
                    res(value);
                },
                (err) => {
                    rej(err);
                }
            )
        })
    })
}
