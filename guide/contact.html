# 迭代器

# 基本用法

function *create() {
    let first = yield 1;
    let second = yield first + 2;
    yield second;
}
let iterator = create(); //生成一个可迭代的对象，是包含一个next的属性
iterator.next();  //返回{value: 1, done: false}  but first's value 取决于第二次next的入参
iterator.next(2); //把2给first; 返回{value: 4, done: false}; 
iterator.next(5); //把4给second； 返回{value: 5, done: false}
iterator.next();  //yield 被执行完毕，返回{value: undefined, done: true}

# 注意点

  1. *func是一个生成器，默认给[Symbol.iterator]属性赋值，是一个具有next属性的对象的函数；
  2. *func() => 生成一个具有next属性的对象
  3. 其*func 等待next的调用，
  4. yield 会保留一个内部指针，用来下次被执行时从这开始
  5. yield 不会超过迭代器边界
let a = 6;
function fc() {
    console.log('等待执行')
    a = 8;
    return a
}
function *f() {
    let result = a + (yield fc()); //6 + undefined; 第二个next入参是undefined
    yield result; 
}
let iterator = f()
iterator.next() //{value: 8, done: false}
iterator.next() //{value: NaN, done: false}
iterator.next() //{value: undefined, done:false}

# 在迭代器中return和抛出错误

看例子

function *f() {
    yield 1;
    return 3; //done为true
}
let iterator = f()
iterator.next()  //{value: 1, done: false}
iterator.next()  //{value: 3, done: true} 这边是区别点
iterator.next()  //{value: undefined, done: true}

个人觉得可以把return理解为只是done为true，value有值的特殊情况

function *f() {
    yield 1;
    return 3; //在这直接结束执行
}
let iterator = f()
iterator.next()  //{value: 1, done: false}
iterator.throw(new Error('error'))  //error
iterator.next()  //{value: undefined, done: true}

# 委托生成器

function *f() {
    yield 1;
    return 2; //在这直接结束执行
}

function *f2() {
    yield 3;
    return 4; //在这直接结束执行
}

function *wrap() {
    yield *f()
    yield *f2()
}

let iterator = wrap();
iterator.next() // {value: 1, done: false}
iterator.next() // {value: 3, done: false}

至于没有输出2，4；因为return的值done是true，虽然有值；但可以通过以一个变量取出来

function *wrap() {
    let return1 = yield *f() //先拿到
    yield return1  //再输出
    yield *f2()
}

# 异步的迭代器

核心就是将next方法闭包在回调内部，等回调执行时，将data带入next，执行next(data);这样外部就能拿到这值

function run(taskDef) {
    let task = taskDef() //迭代器
    let result = task.next()
    function step() {
        if (!result.done) {
            if (typeof result.value === 'function) {
                result.value((err, data) => {
                    if (err) {
                        task.throw(err)
                        return
                    } else {
                        result = task.next(data)
                        step()
                    }
                })
            } else {
                result = task.next(result.value)
                step()
            }
        }
    }   
    step()
}

function readFile() {
    return function(callback) {
        fs.readFile(filename, callback)
    }
}
run(function *() {
    let contents = yield readFile('xxx')
    //dosomething
    return contents
})
