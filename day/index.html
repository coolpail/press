# vue路由重置

import Vue from 'vue';
import Router from 'vue-router'

Vue.use(Router);
const createRouter = () => new Router({
    mode: 'history',
    routes: []
})
const router = createRouter(); //生成一个router，作为页面的路由
export function resetRouter () {
    const newRouter = createRouter();
    router.matcher = newRouter.matcher;  //重新复制新的路由匹配数据
}
export default router;  //导出路由

# 自定义组件模仿v-model

<input v-model="inputValue" />
<!-- 等价于 -->
<input :value="inputValue" @change="inputValue = $event.target.value" />

所以我们写这么写

<child :value.sync="inputValue"></child>
<!-- 子组件 -->
<script>
export default {
    props: {
        //props可以设置值得类型，默认值，是否必传以及校验函数
        value: {
            type: [String, Number],
            required: true,
        },
    },
    //用一个变量中转，子组件中就用_value就不会直接修改父组件的值
    //还是通过update:value事件修改value，间接修改_value
    computed: {
        _value: {
            get() {
                return this.value;
            },
            set(val) {
                this.$emit('update:value', val);
            },
        },
    },
};
